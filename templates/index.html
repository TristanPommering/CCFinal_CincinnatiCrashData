// Fetch and populate the vehicle types dropdown async function fetchVehicleTypes() { try { const response = await fetch('/vehicle-types'); const vehicleTypes = await response.json(); if (vehicleTypes.length === 0) { console.error("No vehicle types found."); alert("No vehicle types available in the database."); } const dropdown = document.getElementById('vehicleType'); vehicleTypes.forEach(type => { const option = document.createElement('option'); option.value = type; option.textContent = type; dropdown.appendChild(option); }); console.log("Vehicle types loaded:", vehicleTypes); } catch (error) { console.error("Error fetching vehicle types:", error); } } // Fetch crash data and update the chart async function fetchCrashData(vehicleType = "") { try { const response = await fetch(`/crash-data?vehicle_type=${vehicleType}`); const data = await response.json(); if (data.length === 0) { console.error("No crash data found."); alert("No crash data available for the selected vehicle type."); } if (crashChart) crashChart.destroy(); const labels = Array.from({ length: 12 }, (_, i) => i + 1); // Months 1-12 const counts = Array(12).fill(0); data.forEach(d => { counts[d.month - 1] = d.count; }); crashChart = new Chart(ctx, { type: 'bar', data: { labels: labels.map(month => `Month ${month}`), datasets: [{ label: 'Number of Crashes', data: counts, backgroundColor: 'rgba(75, 192, 192, 0.2)', borderColor: 'rgba(75, 192, 192, 1)', borderWidth: 1 }] }, options: { responsive: true, scales: { y: { beginAtZero: true } } } }); console.log("Crash data loaded:", data); } catch (error) { console.error("Error fetching crash data:", error); } } // Initialize the dropdown and chart fetchVehicleTypes(); fetchCrashData(); // Update the chart when the dropdown value changes document.getElementById('vehicleType').addEventListener('change', (e) => { const vehicleType = e.target.value; fetchCrashData(vehicleType); });
